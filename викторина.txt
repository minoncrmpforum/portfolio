<script>
// ══════════════════════════════════════════
//  GAME STATE
// ══════════════════════════════════════════
const PLAYER_COLORS = [
  '#00f5ff', '#ff006e', '#ffe600', '#39ff14',
  '#bf5fff', '#ff6b35', '#00ffa3', '#ff3cac'
];

const RANK_EMOJI = ['🥇', '🥈', '🥉', '4️⃣', '5️⃣', '6️⃣', '7️⃣', '8️⃣'];

let state = {
  players: [],
  buzzedPlayer: null,
  selectedScoreTarget: null,
  questionCount: 0,
};

// ══════════════════════════════════════════
//  SETUP
// ══════════════════════════════════════════
let playerCount = 4;

function renderPlayerInputs() {
  const list = document.getElementById('players-input-list');
  list.innerHTML = '';
  for (let i = 0; i < playerCount; i++) {
    const row = document.createElement('div');
    row.className = 'player-input-row';
    row.innerHTML = `
      <span class="player-num">#${i+1}</span>
      <div class="color-dot" style="background:${PLAYER_COLORS[i]}"></div>
      <input type="text" placeholder="Ник игрока ${i+1}" 
             id="player-name-${i}" maxlength="16"
             value="${['XenoS', 'Ghost', 'Viper', 'Luna', 'Kira', 'Raze', 'Crypt', 'Echo'][i] || ''}">
    `;
    list.appendChild(row);
  }
  document.getElementById('player-count-display').textContent = playerCount;
}

function changeCount(delta) {
  playerCount = Math.max(2, Math.min(8, playerCount + delta));
  renderPlayerInputs();
}

function startGame() {
  const names = [];
  for (let i = 0; i < playerCount; i++) {
    const inp = document.getElementById(`player-name-${i}`);
    names.push(inp.value.trim() || `Игрок ${i+1}`);
  }

  state.players = names.map((name, i) => ({
    id: i,
    name,
    score: 0,
    color: PLAYER_COLORS[i],
  }));
  state.buzzedPlayer = null;
  state.selectedScoreTarget = null;
  state.questionCount = 0;

  buildPodiums();
  buildScoringRow();
  showScreen('game-screen');
  updateBuzzStatus(null);
}

// ══════════════════════════════════════════
//  PODIUMS
// ══════════════════════════════════════════
function buildPodiums() {
  const grid = document.getElementById('podium-grid');
  grid.innerHTML = '';

  state.players.forEach(p => {
    const card = document.createElement('div');
    card.className = 'podium-card';
    card.id = `podium-${p.id}`;
    card.style.setProperty('--player-color', p.color);

    card.innerHTML = `
      <span class="podium-rank" id="rank-${p.id}">#1</span>
      <div class="podium-avatar">${p.name.charAt(0).toUpperCase()}</div>
      <div class="podium-name">${p.name}</div>
      <div class="podium-score" id="score-${p.id}">0</div>
      <div class="podium-pts-label">PTS</div>
      <button class="buzz-btn" id="buzz-btn-${p.id}" onclick="playerBuzz(${p.id})">⚡ BUZZ</button>
    `;
    grid.appendChild(card);
  });

  updateRanks();
}

function buildScoringRow() {
  const row = document.getElementById('scoring-row');
  row.innerHTML = '<span class="scoring-label">НАЧИСЛИТЬ → </span>';
  state.players.forEach(p => {
    const btn = document.createElement('button');
    btn.className = 'score-player-btn';
    btn.id = `score-target-btn-${p.id}`;
    btn.style.setProperty('--player-color', p.color);
    btn.innerHTML = `<span style="width:8px;height:8px;border-radius:50%;background:${p.color};display:inline-block"></span> ${p.name}`;
    btn.onclick = () => selectScoreTarget(p.id);
    row.appendChild(btn);
  });

  if (state.players.length > 0) {
    selectScoreTarget(state.players[0].id);
  }
}

function selectScoreTarget(id) {
  state.selectedScoreTarget = id;
  state.players.forEach(p => {
    const btn = document.getElementById(`score-target-btn-${p.id}`);
    if (!btn) return;
    if (p.id === id) {
      btn.style.background = `color-mix(in srgb, ${p.color} 20%, transparent)`;
      btn.style.boxShadow = `0 0 12px color-mix(in srgb, ${p.color} 30%, transparent)`;
    } else {
      btn.style.background = 'transparent';
      btn.style.boxShadow = 'none';
    }
  });
}

// ══════════════════════════════════════════
//  BUZZ SYSTEM
// ══════════════════════════════════════════
function playerBuzz(playerId) {
  if (state.buzzedPlayer !== null) return; // already buzzed

  state.buzzedPlayer = playerId;
  const p = state.players.find(x => x.id === playerId);

  updateBuzzStatus(p);
  selectScoreTarget(playerId);

  // highlight card
  const card = document.getElementById(`podium-${playerId}`);
  card.classList.add('buzzed-card');

  // highlight buzz button
  const btn = document.getElementById(`buzz-btn-${playerId}`);
  btn.classList.add('is-buzzed');
  btn.textContent = '⚡ ОТВЕЧАЕТ!';

  // disable all other buzz buttons
  state.players.forEach(x => {
    if (x.id !== playerId) {
      document.getElementById(`buzz-btn-${x.id}`).disabled = true;
    }
  });

  addLog(`<span class="neutral">${p.name} нажал(а) BUZZ!</span>`);
}

function resetBuzz() {
  state.buzzedPlayer = null;
  state.players.forEach(p => {
    const card = document.getElementById(`podium-${p.id}`);
    card.classList.remove('buzzed-card');
    const btn = document.getElementById(`buzz-btn-${p.id}`);
    btn.classList.remove('is-buzzed');
    btn.disabled = false;
    btn.textContent = '⚡ BUZZ';
  });
  updateBuzzStatus(null);
}

function updateBuzzStatus(player) {
  const bar = document.getElementById('buzz-status-bar');
  if (!player) {
    bar.textContent = '⏳ ЖМИТЕ КНОПКУ ЧТОБЫ ОТВЕТИТЬ';
    bar.classList.remove('buzzed');
  } else {
    bar.textContent = `⚡ ${player.name.toUpperCase()} ОТВЕЧАЕТ!`;
    bar.classList.add('buzzed');
  }
}

// ══════════════════════════════════════════
//  QUESTIONS
// ══════════════════════════════════════════
function publishQuestion() {
  const qText = document.getElementById('question-input').value.trim();
  const cat = document.getElementById('category-input').value.trim();
  if (!qText) return;

  state.questionCount++;
  document.getElementById('round-counter').textContent = `ВОПРОС #${state.questionCount}`;
  document.getElementById('q-category-display').textContent = cat || '— КАТЕГОРИЯ —';

  const display = document.getElementById('question-display');
  display.textContent = qText;
  display.classList.remove('empty');

  resetBuzz();
  addLog(`<span class="neutral">Вопрос #${state.questionCount}: ${qText.substring(0,60)}${qText.length > 60 ? '…' : ''}</span>`);
}

function clearQuestion() {
  document.getElementById('question-input').value = '';
  document.getElementById('question-display').textContent = 'Ведущий ещё не задал вопрос...';
  document.getElementById('question-display').classList.add('empty');
  document.getElementById('q-category-display').textContent = '— КАТЕГОРИЯ —';
  resetBuzz();
}

// ══════════════════════════════════════════
//  SCORING
// ══════════════════════════════════════════
function awardPoints(isCorrect) {
  const pts = isCorrect ? 100 : -50;
  applyScore(pts, isCorrect);
}

function awardCustom() {
  const pts = parseInt(document.getElementById('pts-input').value) || 0;
  applyScore(pts, pts >= 0);
}

function applyScore(pts, isPositive) {
  const id = state.selectedScoreTarget;
  if (id === null) return;
  const p = state.players.find(x => x.id === id);
  if (!p) return;

  p.score += pts;

  const scoreEl = document.getElementById(`score-${id}`);
  scoreEl.textContent = p.score;
  scoreEl.classList.remove('pop');
  void scoreEl.offsetWidth;
  scoreEl.classList.add('pop');

  const card = document.getElementById(`podium-${id}`);
  card.classList.add(isPositive ? 'correct-flash' : 'wrong-flash');
  setTimeout(() => card.classList.remove('correct-flash', 'wrong-flash'), 700);

  const sign = pts >= 0 ? '+' : '';
  const cls = isPositive ? 'pos' : 'neg';
  addLog(`<span class="${cls}">${p.name}: ${sign}${pts} pts → ${p.score} pts</span>`);

  updateRanks();
  resetBuzz();
}

function updateRanks() {
  const sorted = [...state.players].sort((a, b) => b.score - a.score);
  sorted.forEach((p, i) => {
    const el = document.getElementById(`rank-${p.id}`);
    if (el) el.textContent = `#${i+1}`;
  });
}

// ══════════════════════════════════════════
//  LOG
// ══════════════════════════════════════════
function addLog(html) {
  const log = document.getElementById('event-log');
  const entry = document.createElement('div');
  entry.className = 'log-entry';
  const now = new Date();
  const time = `${now.getHours().toString().padStart(2,'0')}:${now.getMinutes().toString().padStart(2,'0')}:${now.getSeconds().toString().padStart(2,'0')}`;
  entry.innerHTML = `<span style="color:#2a3a60">[${time}]</span> ${html}`;
  log.appendChild(entry);
  log.scrollTop = log.scrollHeight;
}

// ══════════════════════════════════════════
//  RESULTS
// ══════════════════════════════════════════
function showResults() {
  const sorted = [...state.players].sort((a, b) => b.score - a.score);

  // PODIUM (top 3)
  const podiumEl = document.getElementById('results-podium');
  podiumEl.innerHTML = '';

  const order = sorted.length >= 3
    ? [sorted[1], sorted[0], sorted[2]]  // silver, gold, bronze layout
    : sorted;

  order.forEach((p, visIdx) => {
    const rank = sorted.findIndex(x => x.id === p.id);
    const card = document.createElement('div');
    card.className = `result-card rank-${rank + 1}`;
    card.style.setProperty('--player-color', p.color);

    const medal = rank < 3 ? RANK_EMOJI[rank] : '';
    card.innerHTML = `
      ${medal ? `<div class="result-rank-badge">${medal}</div>` : ''}
      <div class="result-name">${p.name}</div>
      <div class="result-score">${p.score}</div>
      <div class="result-pts">PTS</div>
    `;
    podiumEl.appendChild(card);
  });

  // FULL LIST
  const listEl = document.getElementById('results-list');
  listEl.innerHTML = '';
  sorted.forEach((p, i) => {
    const row = document.createElement('div');
    row.className = 'result-row';
    row.style.setProperty('--player-color', p.color);
    row.innerHTML = `
      <span class="result-row-rank">${RANK_EMOJI[i] || `#${i+1}`}</span>
      <div class="result-row-dot" style="background:${p.color}"></div>
      <span class="result-row-name">${p.name}</span>
      <span class="result-row-score">${p.score}</span>
    `;
    listEl.appendChild(row);
  });

  showScreen('results-screen');
}

}

  window.scrollTo(0, 0);
}

function confirmReset() {
  document.getElementById('modal-overlay').classList.add('show');
}

  document.getElementById('modal-overlay').classList.remove('show');
}

function resetGame() {
  closeModal();
  showScreen('setup-screen');
}

// ══════════════════════════════════════════
//  KEYBOARD SHORTCUTS
// ══════════════════════════════════════════
document.addEventListener('keydown', (e) => {
  const gameActive = document.getElementById('game-screen').classList.contains('active');
  if (!gameActive) return;

  if (e.target.tagName === 'INPUT') return;

  if (e.code === 'Space') {
    e.preventDefault();
    publishQuestion();
  }
  if (e.key.toLowerCase() === 'r') {
    resetBuzz();
  }
  const num = parseInt(e.key);
  if (num >= 1 && num <= state.players.length) {
    playerBuzz(num - 1);
  }
});

// ══════════════════════════════════════════
//  PARTICLES
// ══════════════════════════════════════════
function spawnParticles() {
  const container = document.getElementById('particles');
  const colors = ['#00f5ff', '#ff006e', '#ffe600', '#bf5fff', '#39ff14'];

  setInterval(() => {
    const p = document.createElement('div');
    p.className = 'particle';
    const size = Math.random() * 4 + 1;
    p.style.cssText = `
      width: ${size}px;
      height: ${size}px;
      left: ${Math.random() * 100}vw;
      background: ${colors[Math.floor(Math.random() * colors.length)]};
      animation-duration: ${Math.random() * 8 + 6}s;
      animation-delay: ${Math.random() * 2}s;
      box-shadow: 0 0 ${size * 3}px currentColor;
    `;
    container.appendChild(p);
    setTimeout(() => p.remove(), 12000);
  }, 400);
}
